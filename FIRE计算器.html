const RETURN_RATE = 0.07; // 7% 年化投资收益率
const WITHDRAW_RATE = 0.04; // 4% 安全提取率
const INFLATION = 0.03; // 3% 通货膨胀率
// ===== Utility Functions =====
function formatMoney(value) {
// 添加千分位分隔符
return '¥' + value.toLocaleString('zh-CN', {
minimumFractionDigits: 0,
maximumFractionDigits: 0
});
}

function formatMoneyWan(value) {
// 用"万"为单位显示
if (value >= 10000) {
return (value / 10000).toFixed(1) + '万';
}
return '¥' + value.toLocaleString('zh-CN');
}

function formatAge(years) {
return years <= 0 ? '已达成' : years + ' 年';
}

// ===== Amount Input Formatting =====
function formatNumberWithCommas(value) {
// 将数字格式化为带千分位分隔符的字符串
if (!value) return '';
const num = parseFloat(value);
if (isNaN(num)) return '';
return num.toLocaleString('zh-CN', {
minimumFractionDigits: 0,
maximumFractionDigits: 0
});
}

function parseFormattedNumber(id, defaultValue = 0) {
// 解析带千分位分隔符的字符串为数字
const input = document.getElementById(id);
if (!input) return defaultValue;
const str = input.value;
if (!str) return defaultValue;
// 移除所有非数字字符（除了小数点和负号）
const cleaned = str.replace(/[^\d.-]/g, '');
const num = parseFloat(cleaned);
return isNaN(num) ? defaultValue : num;
}

function initAmountInputs() {
// 获取所有金额输入框
const amountInputs = document.querySelectorAll('input.with-prefix');
amountInputs.forEach(input => {
// 初始化时格式化显示值
const initialValue = input.value;
if (initialValue) {
input.value = formatNumberWithCommas(initialValue);
}

// 输入时实时更新格式化
input.addEventListener('input', function(e) {
// 获取光标位置
const cursorPos = this.selectionStart;
const oldValue = this.value;
// 解析当前值（直接解析字符串）
const cleaned = this.value.replace(/[^\d.-]/g, '');
const numValue = parseFloat(cleaned) || 0;
// 格式化新值
const formattedValue = formatNumberWithCommas(numValue);
// 更新显示值
this.value = formattedValue;
// 调整光标位置（考虑新增的逗号）
const oldCommas = (oldValue.match(/,/g) || []).length;
const newCommas = (formattedValue.match(/,/g) || []).length;
const newCursorPos = cursorPos + (newCommas - oldCommas);
// 设置光标位置
this.setSelectionRange(newCursorPos, newCursorPos);
});

// 失去焦点时格式化
input.addEventListener('blur', function() {
const cleaned = this.value.replace(/[^\d.-]/g, '');
const numValue = parseFloat(cleaned) || 0;
this.value = formatNumberWithCommas(numValue);
});

// 获得焦点时移除格式化（方便编辑）
input.addEventListener('focus', function() {
const cleaned = this.value.replace(/[^\d.-]/g, '');
const numValue = parseFloat(cleaned) || 0;
this.value = numValue;
});
});
}

// ===== Get Income for Year =====
function getIncomeForYear(year, baseYear) {
const income1 = parseFormattedNumber('income1', 300000);
const income2 = parseFormattedNumber('income2', 400000);
const income3 = parseFormattedNumber('income3', 350000);
const growth1 = parseFloat(document.getElementById('growth1').value) / 100 || 0.05;
const growth2 = parseFloat(document.getElementById('growth2').value) / 100 || 0.03;
const growth3 = parseFloat(document.getElementById('growth3').value) / 100 || 0;

const yearsFromBase = year - baseYear;
if (yearsFromBase < 5) {
// Stage 1: 0-5 years
return income1 * Math.pow(1 + growth1, yearsFromBase);
} else if (yearsFromBase < 10) {
// Stage 2: 5-10 years
const yearsInStage2 = yearsFromBase - 5;
return income2 * Math.pow(1 + growth2, yearsInStage2);
} else {
// Stage 3: 10+ years
const yearsInStage3 = yearsFromBase - 10;
return income3 * Math.pow(1 + growth3, yearsInStage3);
}
}

// ===== Calculate Mortgage =====
function calculateMortgage(principal, years, rate) {
const monthlyRate = rate / 12;
const months = years * 12;
const monthlyPayment = principal * monthlyRate * Math.pow(1 + monthlyRate, months) / (Math.pow(1 + monthlyRate, months) - 1);
const totalPayment = monthlyPayment * months;
const totalInterest = totalPayment - principal;
return {
monthlyPayment: monthlyPayment,
annualPayment: monthlyPayment * 12,
totalInterest: totalInterest
};
}

// ===== Main Calculation =====
function calculate() {
try {
// Get basic inputs
const age = parseFloat(document.getElementById('age').value) || 30;
const savings = parseFormattedNumber('savings', 500000);
const children = parseInt(document.getElementById('children').value) || 0;
// Get children ages (multiple inputs)
const childrenAgeInputs = document.querySelectorAll('.children-age-input');
const childrenAges = Array.from(childrenAgeInputs).map(input => parseInt(input.value) || 5);

// Get expense inputs
const basicExpense = parseFormattedNumber('basicExpense', 80000);
const educationExpense = parseFormattedNumber('educationExpense', 20000);
const housingExpense = parseFormattedNumber('housingExpense', 48000);
const otherExpense = parseFormattedNumber('otherExpense', 20000);

// Get house plan
const housePlan = document.getElementById('housePlan').value;
const houseYear = housePlan === 'planned' ? (parseFloat(document.getElementById('houseYear').value) || 3) : 0;
const houseBudget = parseFormattedNumber('houseBudget', 3000000);
const downPaymentRatio = parseFloat(document.getElementById('downPaymentRatio').value) / 100 || 0.3;
const loanYears = parseFloat(document.getElementById('loanYears').value) || 30;
const loanRate = parseFloat(document.getElementById('loanRate').value) / 100 || 0.042;

// Calculate total expense
let totalExpense = basicExpense + educationExpense + housingExpense + otherExpense;
// Calculate FIRE target
const fireTarget = totalExpense / WITHDRAW_RATE;
// Calculate mortgage if planning to buy
let downPayment = 0;
let mortgagePayment = { monthlyPayment: 0, annualPayment: 0, totalInterest: 0 };
if (housePlan === 'planned') {
downPayment = houseBudget * downPaymentRatio;
const loanAmount = houseBudget - downPayment;
mortgagePayment = calculateMortgage(loanAmount, loanYears, loanRate);
}

// Calculate education cost until children reach 18
let totalEducationCost = 0;
if (children > 0 && childrenAges.length > 0) {
childrenAges.forEach(childAge => {
const yearsTo18 = Math.max(0, 18 - childAge);
totalEducationCost += educationExpense * yearsTo18;
});
}

// Simulate year by year
const startYear = 2024;
let currentWealth = savings;
let yearsToFire = 0;
let fireYear = null;
const maxYears = 50;
const wealthData = [];
const years = [];
let yearsWithoutHouse = 0;
let yearsWithHouse = 0;
const realReturnRate = RETURN_RATE - INFLATION;

for (let i = 0; i <= maxYears; i++) {
const currentAge = age + i;
years.push(currentAge);
// Get income for this year
const yearIncome = getIncomeForYear(i, 0);
// Calculate expense for this year
let yearExpense = basicExpense + otherExpense;
// Add housing expense
if (housePlan === 'planned' && i >= houseYear) {
// After buying house, add mortgage payment
if (i < houseYear + loanYears) {
yearExpense += mortgagePayment.annualPayment;
}
} else {
// Before buying or no plan, use current housing expense
yearExpense += housingExpense;
}
// Add education expense (only until children reach 18)
if (children > 0 && childrenAges.length > 0) {
let educationThisYear = 0;
childrenAges.forEach(childAge => {
const yearsUntil18 = 18 - childAge;
if (i < yearsUntil18) {
educationThisYear += educationExpense;
}
});
yearExpense += educationThisYear;
}
// Deduct down payment in the year of buying
if (housePlan === 'planned' && i === houseYear) {
currentWealth -= downPayment;
}
// Calculate saving for this year
const yearSaving = yearIncome - yearExpense;
// Apply compound interest
currentWealth = currentWealth * (1 + realReturnRate) + yearSaving;
wealthData.push(Math.max(0, Math.round(currentWealth)));
// Check if FIRE achieved
if (!fireYear && currentWealth >= fireTarget) {
fireYear = i;
yearsToFire = i;
yearsWithHouse = fireYear;
}
}

// Calculate without house for comparison
if (housePlan === 'planned') {
let wealthNoHouse = savings;
for (let i = 0; i <= maxYears; i++) {
const yearIncome = getIncomeForYear(i, 0);
let yearExpense = basicExpense + housingExpense + otherExpense;
if (children > 0 && childrenAges.length > 0) {
let educationThisYear = 0;
childrenAges.forEach(childAge => {
const yearsUntil18 = 18 - childAge;
if (i < yearsUntil18) {
educationThisYear += educationExpense;
}
});
yearExpense += educationThisYear;
}
const yearSaving = yearIncome - yearExpense;
wealthNoHouse = wealthNoHouse * (1 + realReturnRate) + yearSaving;
if (wealthNoHouse >= fireTarget) {
yearsWithoutHouse = i;
break;
}
}
}

// Calculate risk level
const avgIncome = (getIncomeForYear(0, 0) + getIncomeForYear(5, 0) + getIncomeForYear(10, 0)) / 3;
const incomeVariance = Math.abs(getIncomeForYear(10, 0) - getIncomeForYear(0, 0)) / avgIncome;
let riskLevel = '低';
if (incomeVariance > 0.5) riskLevel = '高';
else if (incomeVariance > 0.2) riskLevel = '中等';

// Update results
const fireAge = yearsToFire > 0 ? age + yearsToFire : age;
document.getElementById('fireTarget').textContent = formatMoney(fireTarget);
document.getElementById('fireAge').textContent = yearsToFire > 0 ? fireAge + ' 岁' : '已达成';
document.getElementById('riskLevel').textContent = riskLevel;
const avgExpense = totalExpense;
const avgIncome0 = getIncomeForYear(0, 0);
const avgSaving = avgIncome0 - avgExpense;
const savingRate = avgSaving / avgIncome0;
document.getElementById('annualSaving').textContent = formatMoney(Math.max(0, avgSaving));
document.getElementById('savingRate').textContent = (savingRate * 100).toFixed(1) + '%';
document.getElementById('yearsToFire').textContent = formatAge(yearsToFire);
document.getElementById('gapAmount').textContent = formatMoney(Math.max(0, fireTarget - savings));
const houseImpact = housePlan === 'planned' ? (yearsWithHouse - yearsWithoutHouse) : 0;
document.getElementById('houseImpact').textContent = houseImpact >= 0 ? '+' + houseImpact + ' 年' : houseImpact + ' 年';
document.getElementById('totalEducation').textContent = formatMoney(totalEducationCost);

// Update chart
generateChart(years, wealthData, fireTarget, age);
} catch (error) {
console.error('计算错误:', error);
alert('计算出错: ' + error.message);
}
}

// ===== Share Functionality =====
function generateShareUrl() {
const params = new URLSearchParams();
const inputs = document.querySelectorAll('input[type="number"], select');
inputs.forEach(input => {
if (input.id) params.set(input.id, input.value);
});
return window.location.origin + window.location.pathname + '?' + params.toString();
}

function shareLink() {
const url = generateShareUrl();
if (navigator.clipboard) {
navigator.clipboard.writeText(url).then(() => showToast('链接已复制到剪贴板！'));
} else {
const input = document.createElement('input');
input.value = url;
document.body.appendChild(input);
input.select();
document.execCommand('copy');
document.body.removeChild(input);
showToast('链接已复制到剪贴板！');
}
}
